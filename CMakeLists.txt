project(my_online_ik LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)

target_link_libraries(online_ik_test
    PRIVATE
        rtb::Common
        rtb::Utilities

        rtb::InverseDynamics
        rtb::SignalProcessing
        rtb::InputDevice
        rtb::OutputDevice

        rtb::Concurrency

        osimSimulation
        osimCommon
        osimTools
        osimAnalyses
        osimActuators

        SimTKsimbody
        SimTKmath
        SimTKcommon

        pthread
)
